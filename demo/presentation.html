🐍 Python SDK


import asyncio
from ml_platform import Client

async def main():
    async with Client("http://localhost:8000") as c:
        # 注册
        user = await c.auth.register(
            "alice", "alice@test.com", "secret123")
        c.set_token(user["access_token"])

        # 上传 & 训练
        ds = await c.data.upload("iris.csv")
        exp = await c.experiments.create(
            name="demo", dataset_id=ds["id"],
            target_column="species",
            problem_type="classification")
        await c.experiments.run_sklearn(exp["id"])


        # 注册模型 & 部署
        model = await c.models.register(
            "iris-model", exp["id"])
        dep = await c.deployments.create(model["id"])

        # 在线预测
        pred = await c.deployments.predict(
            dep["id"], {
                "sepal_length": 5.1,
                "sepal_width": 3.5,
                "petal_length": 1.4,
                "petal_width": 0.2,
            })
        print(pred)  # {"prediction": ["setosa"]}

asyncio.run(main())

全异步 (httpx + asyncio) · 上下文管理器 · 自动错误处理  |  也可以 client.set_token() 复用 token 而不进入 context manager
